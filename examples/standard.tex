\subsection{Sum of Squares}
\begin{frame}{Let's Prove Something}
\only<1>{
\begin{theorem}[Sum of Squares]
For $n \geq 0$, \[ \sum^n_{i=1} i = \frac{n (n+1)}{2}. \]
\end{theorem}
}
\only<2>{
\begin{proof}
\begin{description}[1in]
\item[Basis:] For $n = 0$, \[ \sum^n_{i=1} i = 0 = \frac{n (n+1)}{2} \]
\item[Recurrence:] Assume the result holds for $n$. Then for $n + 1$,
\[ \sum^{n+1}_{i=1} i
= (n+1) + \sum^n_{i=1} i
= (n+1) + \frac{n(n+1)}{2}
= \frac{(n+1)(n+2)}{2}.
\]
\end{description}
\end{proof}
}
\end{frame}

\subsection{Some Definitions}
\begin{frame}{Some Definitions}
\begin{definition}[Big-$\mathcal{O}$]
A function $f(n)$ is asymptotically bounded below by function $g(n)$, i.e., $f \in \mathcal{O}(g)$, if for some $n_0$ and $c > 0$, $f(n) \geq c \cdot g(n)$ for all $n \geq n_0$.
\end{definition}
\begin{itemize}
\item We say that $f$ is in Big-$\mathcal{O}$ of $g$, $f = \text{Big-}\mathcal{O}(g)$, or more rarely $f \in \text{Big-}\mathcal{O}(g)$; although I would argue that the last option is the most correct.
\end{itemize}
\end{frame}
